brfss = p8105.datasets::brfss_smart2010 %>% 
  janitor::clean_names() %>%
  rename(state = locationabbr, 
         county = locationdesc) %>% 
  filter(topic == "Overall Health") %>% 
  arrange(year, state, county) %>% 
  mutate(as.factor(response),
            fct_relevel(response, 
                        c('Excellent', 'Very good', 'Good', 'Fair', 'Poor')))
